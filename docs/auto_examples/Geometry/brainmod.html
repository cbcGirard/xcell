def hull(mesh):
    hull = ConvexHull(np.array(mesh.points))
    faces = np.column_stack(
        (3*np.ones((len(hull.simplices), 1), dtype=int), hull.simplices)).flatten()
    poly = pv.PolyData(hull.points, faces)
    return poly
